-- Criação do banco de dados
CREATE DATABASE IF NOT EXISTS kuba_tech;
USE kuba_tech;

-- Tabela de Clientes
CREATE TABLE customers (
    cpf VARCHAR(14) PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    phone VARCHAR(15) NOT NULL,
    email VARCHAR(100) NOT NULL
);

-- Tabela de Dispositivos
CREATE TABLE devices (
    serial_number VARCHAR(100) PRIMARY KEY,
    customer_cpf VARCHAR(14) NOT NULL,
    type VARCHAR(50) NOT NULL,
    FOREIGN KEY (customer_cpf) REFERENCES customers(cpf) ON DELETE CASCADE ON UPDATE CASCADE
);

-- Tabela de Ordens de Serviço
CREATE TABLE service_orders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    customer_cpf VARCHAR(14) NOT NULL,
    device_serial VARCHAR(100) NOT NULL,
    technician VARCHAR(100) NOT NULL,
    opening_date DATE NOT NULL,
    status ENUM('A Realizar', 'Em Andamento', 'Finalizada') DEFAULT 'A Realizar',
    FOREIGN KEY (customer_cpf) REFERENCES customers(cpf) ON DELETE CASCADE ON UPDATE CASCADE,
    FOREIGN KEY (device_serial) REFERENCES devices(serial_number) ON DELETE CASCADE ON UPDATE CASCADE
);
